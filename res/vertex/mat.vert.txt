#version 330 core 

// Position/ Co-ordinates 
layout (location = 0) in vec3 aPos; 
// Colours 
layout (location = 1) in vec3 aColour; 
// Textures 
layout (location = 2) in vec2 aTex; 

// Outputs the colour for the fragment shader 
out vec3 colour; 

out vec2 texCoord; 

uniform mat4 model; 
uniform mat4 view; 
uniform mat4 proj; 


void main()
{
   gl_Position = proj * view * model * vec4(aPos, 1.0f); 
   colour = aColour; 
   texCoord = aTex; 
}
